round
An implementation of a call to the round api. 
                    #define MODULE_MATH 4
                    #define MATH_ROUND  0

                    proc round(number n)
                        struct Round
                            number result
                        ends
                        Round round
                        set      round.result, n
                        addr     round
                        module   MODULE_MATH,  MATH_ROUND
                        return   round.result
                    endp
