sleep

An implementation of a call to the sleep api. 
                    #define MODULE_STANDARD 1
                    #define STANDARD_SLEEP  2

                    proc sleep(number milliseconds)
                        struct Sleep
                            number milliseconds
                        ends
                        Sleep sleep
                        set    sleep.milliseconds, milliseconds
                        addr   sleep
                        module MODULE_STANDARD, STANDARD_SLEEP
                    endp
